--- recusive function to make binary search
function bn2(a, t) {
  let mid = Math.floor((a.length - 1) / 2);
  if (a[mid] === t) {
    return a[mid];
  } else if (a[mid] < t) {
    return bn2(a.splice(1), t);
  } else {
    return bn2(a.splice(a.length - 1, 1), t);
  }
}
